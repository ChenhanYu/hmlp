template<typename T>
class hmlp::gofmm::Factor< T >

for each level for each alpha

if ( leaf )

LL' = Chol( Kaa ) U = inv( L ) * proj' or U' = proj * inv( L' ) QR = qr( U ) or LQ' = lq( U' )

else

LL' = Chol( I + k.R * C * k.R' )

if ( not root )

U = inv( L ) * [ l.R * proj' or r.R ] U' = proj' * [ l.R * inv( L' ) r.R ] QR = qr( U ) or LQ' = lq( U' )
