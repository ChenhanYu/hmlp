84  static inline void squaredDistances( const TP* X, const TP* Y, size_t d, T* K, size_t m, size_t n )
85  {
86  innerProducts( X, Y, d, K, m, n );
87  vector<T> squaredNrmX( m, 0 ), squaredNrmY( n, 0 );
89  #pragma omp parallel for
90  for ( size_t i = 0; i < m; i ++ ) squaredNrmX[ i ] = xdot( d, X + i * d, 1, X + i * d, 1 );
91  #pragma omp parallel for
92  for ( size_t j = 0; j < n; j ++ ) squaredNrmY[ j ] = xdot( d, Y + j * d, 1, Y + j * d, 1 );
93  #pragma omp parallel for collapse(2)
94  for ( size_t j = 0; j < n; j ++ )
95  for ( size_t i = 0; i < m; i ++ )
96  K[ j * m + i ] = squaredNrmX[ i ] - 2 * K[ j * m + i ] + squaredNrmY[ j ];
